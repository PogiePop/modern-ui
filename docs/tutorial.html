全流程实现教程

从零开始构建一个完整的 ModernUI 应用程序。

第1步：创建项目

# 克隆仓库
git clone https://github.com/PogiePop/modern-ui.git
cd modern-ui
mkdir build && cd build
cmake .. -G "Visual Studio 17 2022" -A x64

第2步：编写最小程序

// main.cpp
#include "core/App.hpp"
#include "widgets/Container.hpp"
#include "widgets/Label.hpp"

int main() {
    ui::App app;
    app.init("My App", 800, 600);

    auto label = std::make_unique<ui::Label>("Hello ModernUI!");
    label->setFont(app.getFont());
    label->setFontSize(24);

    auto root = std::make_unique<ui::Container>();
    root->addChild(std::move(label));
    app.setRoot(std::move(root));
    app.run();
}

第3步：添加按钮和布局

auto root = std::make_unique<Container>(LayoutDirection::Vertical);
root->setPadding(EdgeInsets::all(20));
root->setSpacing(12);

auto title = std::make_unique<Label>("Welcome");
title->setFontSize(28); title->setColorRole(ColorRole::Primary);
root->addChild(std::move(title));

auto btn = std::make_unique<Button>("Click Me");
btn->setVariant(Button::Primary); btn->setCornerRadius(8);
btn->setOnClick([](){ printf("Clicked!\n"); });
root->addChild(std::move(btn));

第4步：添加主题切换

Theme dark = Theme::defaultDark();
Theme light = Theme::defaultLight();
ThemeTransition transition;
transition.setImmediate(dark);

auto themeSwitch = std::make_unique<Switch>();
themeSwitch->setChecked(true);
themeSwitch->setOnChanged([&](bool d) {
    transition.setTarget(d ? dark : light, 0.5f);
});

// 每帧驱动动画
struct ThemeDriver : Widget {
    void tickAnimation(float dt) override { transition.tick(dt); }
    Size measure(const Size&) const override { return {0,0}; }
    void paint(Painter&) override {}
};
root->addChild(std::make_unique<ThemeDriver>());

第5步：构建并运行

cmake --build . --config Release
./Release/exp.exe

第6步：发布

# 安装发布文件
cmake --install . --prefix ./install --config Release

# 目录结构:
# install/include/modernui/  -- 头文件
# install/lib/               -- 静态库 + DLL 导入库
# install/bin/               -- DLL
