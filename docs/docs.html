API 文档

App — 应用程序入口

ui::App app;
app.init("窗口标题", 800, 600);   // 创建窗口 + 初始化 OpenGL
app.setRoot(std::move(root));     // 设置根控件树
app.run();                        // 进入主循环
app.quit();                       // 退出

Widget — 控件基类

方法说明
setBounds(Rect)设置位置和大小
setVisible(bool)控制可见性
setTheme(Theme*)设置主题指针
setIcon(IconType)设置图标
setShadow(Point,float,Color)设置阴影
setOpacity(float)设置不透明度
fadeIn/fadeOut(float)淡入淡出动画
addChild/removeChild控件树管理

Container — 布局容器

auto c = std::make_unique(LayoutDirection::Vertical);

c->setSpacing(8);
c->setBackgroundColor(Color::fromHex(0xFF1A1A2E));

c->addChild(std::move(child2));

LayoutDirectionHorizontal, Vertical

AlignStart, Center, End, Stretch, SpaceBetween, SpaceAround, SpaceEvenly

Theme — 主题系统

Theme dark = Theme::defaultDark();
Theme light = Theme::defaultLight();

// 20种颜色角色
dark.setColor(ColorRole::Primary, Color::fromHex(0xFF6366F1));

// 主题切换动画
ThemeTransition transition;
transition.setTarget(light, 0.5f); // 0.5秒动画过渡
transition.tick(dt);               // 每帧调用

Config — 全局配置

auto& cfg = Config::instance();
cfg.multiThreaded = true;
cfg.darkMode = false;
cfg.scrollStep = 50.0f;
cfg.loadFromFile("modernui.cfg");
cfg.saveToFile("modernui.cfg");
