App — 应用程序入口

初始化与运行

ui::App app;
app.init("窗口标题", 800, 600);  // 创建窗口、初始化 OpenGL
app.setRoot(std::move(widget));   // 设置根控件
app.run();                        // 进入主事件循环
app.quit();                       // 退出程序

Overlay / 焦点 / 光标

app.setOverlay(widget);            // 设置顶层覆盖控件（菜单、对话框等）
app.requestFocus(widget);          // 请求焦点
app.setMultiThreaded(true);        // 启用多线程布局

Config — 全局配置

单例访问与配置项

auto& cfg = Config::instance();
cfg.darkMode = false;              // 亮色/暗色主题
cfg.multiThreaded = true;          // 多线程渲染
cfg.themeTransitionDuration = 0.5f;// 主题切换动画时长(秒)
cfg.scrollStep = 50.0f;            // 滚轮步长(px)
cfg.defaultFontSize = 16.0f;       // 默认字号
cfg.animationsEnabled = true;      // 启用动画
cfg.showFPS = false;               // 显示帧率

cfg.loadFromFile("modernui.cfg");  // 从文件加载配置
cfg.saveToFile("modernui.cfg");    // 保存配置到文件
cfg.apply();                       // 同步到运行中子系统

Container — 布局容器

创建与方向

// Row（水平排列）
auto row = std::make_unique<Container>(LayoutDirection::Horizontal);
// Column（垂直排列，默认）
auto col = std::make_unique<Container>(LayoutDirection::Vertical);

row->setDirection(LayoutDirection::Horizontal);
row->setSpacing(8);                        // 子控件间距
row->setPadding(EdgeInsets::all(12));      // 内边距
row->setAlign(Container::Align::Center);   // 对齐方式

子控件尺寸策略

row->setChildSizing(0, Sizing::fixed(100), Sizing::intrinsic());
// Sizing::fixed(px)    — 固定像素
// Sizing::fill(weight) — 弹性填充
// Sizing::intrinsic()  — 自适应内容（默认）

背景色

row->setBackgroundColor(Color::fromHex(0xFF1A1A2E));
row->clearBackgroundColor();

Label — 文本标签

基本用法

auto label = std::make_unique<Label>("Hello World");
label->setFont(font);
label->setFontSize(16);
label->setColorRole(ColorRole::Text);        // 主题色（跟随主题切换动画）
label->setColor(Color::fromHex(0xFFFFFFFF)); // 固定色
label->setTextAlign(TextAlign::Center);      // Left / Center / Right

// 选中与复制
label->selectAll();
label->copy();                               // Ctrl+C
// 右键菜单：Copy / Select All

Button — 按钮

变体与样式

auto btn = std::make_unique<Button>("Click Me");
btn->setVariant(Button::Primary);     // Primary/Secondary/Danger/Success/Ghost
btn->setPill(true);                   // 全圆角药丸形
btn->setCompact(true);                // 紧凑内边距
btn->setCornerRadius(8);              // 圆角半径
btn->setMinWidth(100);                // 最小宽度
btn->setIcon(IconType::Search);       // 图标

// 自定义颜色（覆盖变体色）
btn->setSolidColor(Color::fromHex(0xFF6366F1));
btn->setGradient(topColor, bottomColor);
btn->setHoverColor(hoverColor);
btn->setPressColor(pressColor);

btn->setOnClick([](){ printf("clicked\n"); });
btn->click();                         // 程序触发点击

t.setColor(ColorRole::Success, Color::fromHex(0xFF22C55E));

// 主题切换动画

transition.setTarget(light, 0.5f);      // 0.5秒过渡
